全局引入 #

在main.js中

js
import { createApp } from 'vue';
import App from './App.vue';
// 完整引入组件库
import TinyDesign from '@lin_ui/web-vue';

const app = createApp(App);
// 全局安装
app.use(TinyDesign).mount('#app');
